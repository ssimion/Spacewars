  7 /** Inits the Game Class

 11 function Game() {
 12 	this.PlayerArray = [new Player("Stan"),new Player("Stef")]
 13 	this.turnNumber = 0;
 14 	this.activePlayer = this.PlayerArray[0];
 15 }
 16 /** Init function
 17  * @memberOf Game
 18  * @public
 19  */
 20 Game.prototype.Init =  function (){
 21 	//DrawFiveCards for each player
 22 	for (var i = 0; i < g_NumberofCardsAtStart; ++i){
 23 		for(var j = 0; j < this.PlayerArray.length ; j ++)
 24 			this.PlayerArray[j].DrawCardFromDeck();
 25 	}
 26 	for (var i = 0; i < this.activePlayer.GetNumberOfCardsInHand();++i){
 27 		DrawCardInHand(this.activePlayer.GetCardsInHand()[i],i);
 28 	}
 29 }
 30 /** This function switch between the players.
 31  * @memberOf Game
 32  * @public
 33  */
 34 Game.prototype.NextTurn = function () {
 35 	this.turnNumber++;
 36 }
 37 /** This function returns the active player.
 38  * @memberOf Game
 39  * @public
 40  */
 41 Game.prototype.GetActivePlayer = function () {
 42 	return this.activePlayer;
 43 }
 44 /** This function returns the other player.
 45  * @memberOf Game
 46  * @public
 47  */
 48 Game.prototype.GetInactivePlayer = function(){
 49 	for (var i = 0;i < this.PlayerArray.length; ++i)
 50 		if (this.PlayerArray[i].GetName() != this.activePlayer.GetName())
 51 			return this.PlayerArray[i];
 52 }
 53 /** This function returns the number of the current turn for the player.
 54  * @memberOf Game
 55  * @public
 56  */
 57 Game.prototype.GetCurrentTurn = function () {
 58 	return this.turnNumber;
 59 }
