136 /** This function creates a Card Array that contains the enemy cards 
137  * @memberOf Game
138  * @public
139  */
140 Game.prototype.GetOpposingPlayerBoard = function(){
141 	if (this.previousPlayerMoveString == ""){
142 		return;
143 	}else{
144 		var previousPlayerMoveJSONformat = JSON.parse(this.previousPlayerMoveString);
145 		currentGame.activePlayer.SetHealth(previousPlayerMoveJSONformat.health);
146 		var opposingPlayerBoard = [];
147 		var opPlayerCard;
148 
149 
150 		
151 		// check whether this is a unitCard
152 		var isUnitCard = false; 
153 		
154 		for(var i = 0; i < previousPlayerMoveJSONformat.cards.length; i++)
155 		{
156 			isUnitCard = previousPlayerMoveJSONformat.cards[i].hasOwnProperty("health") ? true : false;
157 			
158 			if(isUnitCard){
159 				opPlayerCard = new UnitCard();
160 			} else {
161 				opPlayerCard = new TrapCard();
162 			}
163 		
164 			opPlayerCard.nameOfCard = previousPlayerMoveJSONformat.cards[i].nameOfCard;
165 			opPlayerCard.descriptionOfCard = previousPlayerMoveJSONformat.cards[i].descriptionOfCard;
166 			opPlayerCard.uniqueID = previousPlayerMoveJSONformat.cards[i].uniqueID;
167 
168 			if(isUnitCard) {
169 				opPlayerCard.health = previousPlayerMoveJSONformat.cards[i].health;
170 				opPlayerCard.attack = previousPlayerMoveJSONformat.cards[i].attack;
171 			} else {
172 				opPlayerCard.type = previousPlayerMoveJSONformat.cards[i].type;
173 				opPlayerCard.effect = previousPlayerMoveJSONformat.cards[i].effect;
174 			}
175 			
176 			opposingPlayerBoard.push(opPlayerCard);
177 		}
178 		
179 		return opposingPlayerBoard;
180 	}
181 }
