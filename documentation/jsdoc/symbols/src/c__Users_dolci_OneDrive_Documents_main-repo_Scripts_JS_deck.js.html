1 "use strict";
  2 var g_types = ["damage","move","destroy"];
  3 
  4 /** Inits the Deck Class
  5  * @class Deck
  6  */
  7 var Deck = function () {
  8 	this.cardList = [];
  9 	for (var i = 0; i < 10; ++i)
 10 	{
 11 		this.cardList.push(new UnitCard(Math.floor(Math.random() * 7),Math.floor(Math.random() * 7)))
 12 		this.cardList.push(new TrapCard(g_types[Math.floor(Math.random() * 3)],Math.floor(Math.random() * 5)))
 13 	}
 14 	this.Shuffle();
 15 }
 16 /** Mixes the cards contained in the deck.
 17  * Taken from http://stackoverflow.com/questions/2450954/how-to-randomize-shuffle-a-javascript-array
 18  * @memberOf Deck
 19  * @public
 20  */
 21 Deck.prototype.Shuffle = function () {
 22 	var array = this.cardList;
 23 	var currentIndex = array.length,
 24 	temporaryValue,
 25 	randomIndex;
 26 	// While there remain elements to shuffle...
 27 	while (0 !== currentIndex) {
 28 		// Pick a remaining element...
 29 		randomIndex = Math.floor(Math.random() * currentIndex);
 30 		currentIndex -= 1;
 31 
 32 		// And swap it with the current element.
 33 		temporaryValue = array[currentIndex];
 34 		array[currentIndex] = array[randomIndex];
 35 		array[randomIndex] = temporaryValue;
 36 	}
 37 	this.cardList = array;
 38 }
 39 /** This function returns the top card of the deck, and removes it from it.
 40  * @memberOf Deck
 41  * @public
 42  */
 43 Deck.prototype.TopCard = function () {
 44 	return this.cardList.pop();
 45 }
 46 /** This function returns the cardlist
 47  * @memberOf Deck
 48  * @public
 49  */
 50 Deck.prototype.GetCardList = function () {
 51 	return this.cardList;
 52 }
 53 /** Returns the number of Cards remaining in the Deck.
 54  * @memberOf Deck
 55  * @public
 56  */
 57 Deck.prototype.GetNumberOfCards = function () {
 58 	return this.cardList.length;
 59 }
