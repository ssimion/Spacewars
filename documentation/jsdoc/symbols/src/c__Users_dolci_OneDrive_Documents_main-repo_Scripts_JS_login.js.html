  2 /** Inits the login Class
  3  * @class login
  4  */
  5 var login = login || {};
  6 
  7 /**
  8  * This function allows us to load up the game service via the discovery doc
  9  * and makes calls directly through the client library instead of needing
 10  * to specify the REST endpoints.
 11  * @memberOf login
 12  * @public
 13  */
 14 login.loadClient = function () {
 15 
 16 	// Load up /games/v1
 17 	gapi.client.load('games', 'v1', function (response) {
 18 		leadManager.preloadData();
 19 	});
 20 
 21 	// Load up v1management
 22 	gapi.client.load('gamesManagement', 'v1management', function (response) {
 23 	});
 24 
 25 	// Load up /plus/v1
 26 	gapi.client.load('plus', 'v1', function (response) {
 27 		console.log(response);
 28 		var request = gapi.client.plus.people.list(
 29 		{
 30 			"collection": "visible",
 31 			"userId": "me"
 32 		});
 33 		
 34 		request.execute(function(response)
 35 		{
 36 			usersFriends = response.items
 37 		});
 38 	});
 39 
 40 };
 41 /**
 42  * @memberOf login
 43  * @public
 44  */
 45 login.handleAuthResult = function (auth) {
 46 	console.log('We are in handle auth result', auth);
 47 	if (auth && auth.error == null) {
 48 		console.log('Hooray! You\'re logged in! ', auth);
 49 		$('#loginDiv').fadeOut();
 50 		login.loadClient();
 51 	} else {
 52 		if (auth /*&& auth.hasOwnProperty('error')*/
 53 		)
 54 		{
 55 			console.log('Login failed because: ', auth.error);
 56 		}
 57 		$('#loginDiv').fadeIn();
 58 	}
 59 };
