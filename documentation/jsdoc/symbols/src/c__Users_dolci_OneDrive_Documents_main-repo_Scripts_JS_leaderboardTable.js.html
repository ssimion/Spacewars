1 "use strict";
  2 /** Inits the leadManager Class
  3  * @class leaderboardTable
  4  */
  5 var leaderboardTable = leaderboardTable || {}
  6 
  7 leaderboardTable.BACK_TO_GAME = 1;
  8 leaderboardTable.BACK_TO_ALL_LEADERBOARDS = 2;
  9 /**
 10  * @memberOf leaderboardTable
 11  * @public 
 12  */
 13 leaderboardTable.goBackTo = leaderboardTable.BACK_TO_ALL_LEADERBOARDS
 14 
 15 	leaderboardTable.showLeaderboard = function (leaderboardId, backDestination) {
 16 	leaderboardTable.goBackTo = backDestination;
 17 	console.log("I am going to show leaderboard ", leaderboardId);
 18 	$('#leaderboardTable tbody').html('');
 19 
 20 	var request = gapi.client.games.scores.list({
 21 			collection : 'SOCIAL',
 22 			leaderboardId : leaderboardId,
 23 			timeSpan : 'WEEKLY'
 24 		});
 25 	request.execute(function (response) {
 26 		console.log('This is your data: ', response);
 27 		if (response.hasOwnProperty('items')) {
 28 			for (var i = 0; i < response.items.length; i++) {
 29 				var $leaderboardRow = leaderboardTable.buildTableRowFromData(response.items[i]);
 30 				$leaderboardRow.appendTo($('#leaderboardTable tbody'));
 31 			}
 32 		}
 33 		$('#leaderboard').fadeIn();
 34 	});
 35 	$('#pageHeader').text(leadManager.getLeaderboardObject(leaderboardId).name);
 36 };
 37 /**
 38  * @memberOf leaderboardTable
 39  * @public 
 40  */
 41 leaderboardTable.buildTableRowFromData = function (rowObj) {
 42 	var scorePlayer = rowObj.player;
 43 	var $tableRow = $('<tr></tr>');
 44 	var $iconCell = $('<td></td>')
 45 		.append($('<img />')
 46 			.prop('src', scorePlayer.avatarImageUrl + '?sz=75'));
 47 	var $nameCell = $('<td></td>').text(scorePlayer.displayName);
 48 	var $scoreCell = $('<td></td>').text(rowObj.formattedScore);
 49 	$tableRow.append($iconCell).append($nameCell).append($scoreCell);
 50 
 51 	return $tableRow;
 52 };
 53 /**
 54  * @memberOf leaderboardTable
 55  * @public 
 56  */
 57 leaderboardTable.goBack = function () {
 58 	$('#leaderboard').hide();
 59 	if (leaderboardTable.goBackTo == leaderboardTable.BACK_TO_ALL_LEADERBOARDS) {
 60 		$('#leaderboards').fadeIn();
 61 	} else {
 62 		$('#game').fadeIn();
 63 	}
 64 };
