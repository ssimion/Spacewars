1 "use strict";
  2 /**
  3  * @class AchievementLoad
  4  * @public
  5  */
  6 var AchievementLoad = AchievementLoad || {};
  7 /**
  8  * @memberOf AchievementLoad
  9  * @public
 10  */
 11 AchievementLoad.achievements = {};
 12 /**
 13  * @memberOf AchievementLoad
 14  * @public
 15  */
 16 AchievementLoad.preloaded = false;
 17 
 18 /**
 19  * @memberOf AchievementLoad
 20  * @public
 21  */
 22 AchievementLoad.loadData = function() {
 23   var request = gapi.client.games.achievementDefinitions.list();
 24   request.execute(function(Info) {
 25     console.log('Achievement definitions', Info);
 26     if (Info.kind == 'games#achievementDefinitionsListResponse' &&
 27         Info.hasOwnProperty('items')) {
 28       for (var i =0; i<Info.items.length; i++) {
 29         AchievementLoad.achievements[Info.items[i].id] = Info.items[i];
 30         AchievementLoad.achievements[Info.items[i].id].achievementState = Info.items[i].initialState;
 31       }
 32       welcome.dataLoaded(welcome.ENUM_ACHIEVEMENT_DEFS);
 33       AchievementLoad.loadAchievementsEarnedByPlayer();
 34 
 35     }
 36   });
 37 };
 38 /**
 39  * @memberOf AchievementLoad
 40  * @public
 41  */
 42 AchievementLoad.clearData = function() {
 43   AchievementLoad.achievements = {};
 44   AchievementLoad.preloaded = false;
 45 };
 46 /**
 47  * @memberOf AchievementLoad
 48  * @public
 49  */
 50 AchievementLoad.loadAchievementsEarnedByPlayer = function() {
 51   var request = gapi.client.games.achievements.list({playerId: 'me'});
 52   request.execute(function(Info) {
 53     console.log('Your achievement data: ', Info);
 54     if (Info.kind == 'games#playerAchievementListResponse' &&
 55         Info.hasOwnProperty('items')) {
 56       for (var i=0; i<Info.items.length; i++) {
 57         var AchievementNext = Info.items[i];
 58         AchievementLoad.achievements[AchievementNext.id].achievementState = AchievementNext.achievementState;
 59         if (AchievementNext.hasOwnProperty('formattedCurrentStepsString')) {
 60           AchievementLoad.achievements[AchievementNext.id].formattedCurrentStepsString = AchievementNext.formattedCurrentStepsString;
 61         }
 62       }
 63       welcome.dataLoaded(welcome.ENUM_ACHIEVEMENT_PROGRESS);
 64     } else {
 65       console.log("**Unexpected response **", Info);
 66     }
 67   });
 68 
 69   AchievementLoad.preloaded = true;
 70 
 71 };
 72 /**
 73  * @memberOf AchievementLoad
 74  * @public
 75  */
 76 AchievementLoad.getNameForId = function(Achievementid)
 77 {
 78   return AchievementLoad.achievements[Achievementid].name;
 79 };
 80 /**
 81  * @memberOf AchievementLoad
 82  * @public
 83  */
 84 AchievementLoad.submitProgress = function(Achievementid, amount)
 85 {
 86   var request = gapi.client.games.achievements.increment(
 87       {achievementId: Achievementid,
 88       stepsToIncrement: amount}
 89   );
 90   request.execute(function(Info) {
 91     console.log('Data from incrementing achievement is ', Info);
 92     // Let's updated our locally cached version
 93     AchievementLoad.achievements[Achievementid].currentSteps = Info.currentSteps;
 94     AchievementLoad.achievements[Achievementid].formattedCurrentStepsString = String(Info.currentSteps);
 95     if (Info.newlyUnlocked ) {
 96       AchPopup.showAchievementWidget(Achievementid);
 97     } else {
 98       console.log('You either haven\'t unlocked ' + AchievementLoad.achievements[Achievementid].name
 99           + ' yet, or you unlocked it already.');
100     }
101   });
102 };
103 /**
104  * @memberOf AchievementLoad
105  * @public
106  */
107 AchievementLoad.unlockAchievement = function(Achievementid)
108 {
109   var request = gapi.client.games.achievements.unlock(
110       {achievementId: Achievementid}
111   );
112   request.execute(function(Info) {
113     console.log('Data from earning achievement is ', Info);
114     if (Info.newlyUnlocked ) {
115       AchPopup.showAchievementWidget(Achievementid);
116       AchievementLoad.loadAchievementsEarnedByPlayer();
117     } else {
118       console.log('You unlocked ' + AchievementLoad.achievements[Achievementid].name + ' but you already unlocked it earlier.');
119     }
120   });
121 
122 };
