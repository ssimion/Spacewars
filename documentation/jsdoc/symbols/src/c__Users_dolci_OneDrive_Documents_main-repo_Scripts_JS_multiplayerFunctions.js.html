 38 /** This function gets the list of the Active matches hosted
 39  * by the Google play Services.
 40  */
 41 function listActiveMatches() {
 42 	var httpRequest = gapi.client.games.turnBasedMatches.list();
 43 
 44 	httpRequest.execute(function (matchListResponse) {
 45 		var gamesCounter = 0;
 46 		for (var i = 0; i < matchListResponse.items.length; i++) {
 47 			if (matchListResponse.items[i].userMatchStatus == "USER_INVITED") {
 48 				listGamesInvitedTo[gamesCounter] = matchListResponse.items[i];
 49 				gamesCounter++;
 50 			}
 51 		}
 52 
 53 	});
 54 }

 80 		});
 81

 83 		console.log("Match status: ", matchDataResponse.userMatchStatus);
 84 		if (matchDataResponse.userMatchStatus == "USER_TURN") {
 85 			userMatchStatus = "USER_TURN";
