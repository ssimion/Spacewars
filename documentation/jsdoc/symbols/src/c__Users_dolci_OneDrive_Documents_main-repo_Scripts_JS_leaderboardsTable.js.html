1 "use strict";
  2 /** Inits the leadManager Class
  3  * @class leaderboardsTable
  4  */
  5 var leaderboardsTable = leaderboardsTable || {};
  6 /**
  7  * @memberOf leaderboardsTable
  8  * @public 
  9  */
 10 leaderboardsTable.showAllLeaderboards = function () {
 11 	console.log("inside showAllLeaderboards");
 12 	leaderboardsTable.clearOut();
 13 	if (leadManager.preloaded) {
 14 		console.log("leadManager.preloaded true");
 15 		$.each(leadManager.leaderboards, function (id, leadObject) {
 16 			var $leaderboardRow = leaderboardsTable.buildLeaderboardsRowFromData(leadObject);
 17 			console.log("built leaderboard from data");
 18 			$leaderboardRow.appendTo($('#leaderboardsTable tbody'));
 19 		});
 20 		$('#leaderboards').fadeIn();
 21 	}
 22 	$('#pageHeader').text('Leaderboards');
 23 };
 24 /**
 25  * @memberOf leaderboardsTable
 26  * @public 
 27  */
 28 leaderboardsTable.buildLeaderboardsRowFromData = function (leadObj) {
 29 	var $tableRow = $('<tr></tr>');
 30 	var leaderboardIcon = (leadObj.iconUrl) ? leadObj.iconUrl : 'img/genericLeaderboard.png';
 31 	var $iconCell = $('<td></td>')
 32 		.append($('<img />')
 33 			.prop('src', leaderboardIcon).addClass('medIcon'));
 34 	var $nameCell = $('<td></td>').text(leadObj.name);
 35 	var $viewButtonCell = $('<td></td>')
 36 		.append($('<input type="button" />')
 37 			.prop('value', 'View')
 38 			.click(function () {
 39 				leaderboardsTable.selectLeaderboard(leadObj.id)
 40 			}));
 41 	$tableRow.append($iconCell).append($nameCell).append($viewButtonCell);
 42 
 43 	return $tableRow;
 44 };
 45 /**
 46  * @memberOf leaderboardsTable
 47  * @public 
 48  */
 49 leaderboardsTable.selectLeaderboard = function (leaderboardId) {
 50 	$('#leaderboards').hide();
 51 	leaderboardTable.showLeaderboard(leaderboardId, leaderboardTable.BACK_TO_ALL_LEADERBOARDS);
 52 
 53 };
 54 /**
 55  * @memberOf leaderboardsTable
 56  * @public 
 57  */
 58 leaderboardsTable.clearOut = function () {
 59 	$('#leaderboardsTable tbody').html('');
 60 };
 61 /**
 62  * @memberOf leaderboardsTable
 63  * @public 
 64  */
 65 leaderboardsTable.goBack = function () {
 66 	$('#leaderboards').fadeOut();
 67 	//welcome.loadUp();
 68 };
