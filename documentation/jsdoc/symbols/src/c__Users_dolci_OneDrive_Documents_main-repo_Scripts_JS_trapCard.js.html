1 "use strict";
  2 
  3 /** Trap card class after being dropped they disappear.
  4  * @class TrapCard
  5  * @extend Card
  6  * @param {string} type: the type ["damage","move","destroy"]
  7  * @param {integer }effect: the effect amount or if "destroy" just don't add it.
  8  */
  9 var TrapCard = function(type,effect) {
 10 	this.nameOfCard = "Trap Card";
 11 	this.type = type || "damage";
 12 	this.uniqueID = Math.round(Math.random() * 1000000000000);
 13 	if (this.type == "destroy"){
 14 		this.effect = "destroy";
 15 		this.descriptionOfCard = "Destroys A Card";
 16 	}
 17 	else if (this.type == "move"){
 18 		this.effect = 2; //Max move is set to one as it would be abused otherwise
 19 		this.descriptionOfCard = "Adds one move for this turn";
 20 	}
 21 	else {
 22 		this.effect = effect || 5;
 23 		this.descriptionOfCard = "Deals "+this.effect+" points of damage";
 24 	}
 25 };
 26 
 27 TrapCard.prototype = new Card();
 28 /** Types can be DirectDamage, Move, Destroy Card. */
 29 /** Returns the health of the unit card
 30  * @memberOf TrapCard
 31  * @public
 32  */
 33 TrapCard.prototype.GetType = function()
 34 {
 35 	return this.type;
 36 }
 37 /** Returns the damage dealt by the card if applicable.
 38  * @memberOf TrapCard
 39  * @public
 40  */
 41 TrapCard.prototype.GetDamage = function()
 42 {
 43 	if (this.type == "damage")
 44 		return this.effect;
 45 	else
 46 		console.log("This is not a Move Bonus Card");
 47 }
 48 /** Returns the extra number of moves granted by the card if applicable
 49  * @memberOf TrapCard
 50  * @public
 51  */
 52 TrapCard.prototype.GetMoveBonus = function()
 53 {
 54 	if (this.type == "Move")
 55 		return this.effect;
 56 	else
 57 		console.log("This is not a Move Bonus Card");
 58 }
 59 /** Override the to String Function
 60  * @memberOf TrapCard
 61  * @public
 62  */
 63 TrapCard.prototype.toString = function(){ 
 64 	return '[TrapCard "' + this.health + '"]';
 65 }
