1 "use strict";
  2 /** Inits the leadManager Class
  3  * @class leaderboardWidget
  4  */
  5 var leaderboardWidget = leaderboardWidget || {};
  6 /**
  7  * @memberOf leaderboardWidget
  8  * @public 
  9  */
 10 leaderboardWidget.show = function (leadId) {
 11 	var leaderboardObject = leadManager.getLeaderboardObject(leadId);
 12 
 13 	// Let's populate the little widget
 14 	var iconUrl = (leaderboardObject.iconUrl) ? leaderboardObject.iconUrl : 'img/genericLeaderboard.png';
 15 	$('#leadUnlocked img').prop('src', iconUrl);
 16 	$('#leadUnlocked #leadName').text(leaderboardObject.name);
 17 	$('#leadUnlocked').css({
 18 		top : '250px',
 19 		opacity : '1.0'
 20 	});
 21 	$('#leadUnlocked').show();
 22 	$('#leadUnlocked').delay(3000).animate({
 23 		top : '50px',
 24 		opacity : 0.1
 25 	}, 500, function () {
 26 		$('#leadUnlocked').hide();
 27 	});
 28 
 29 };
