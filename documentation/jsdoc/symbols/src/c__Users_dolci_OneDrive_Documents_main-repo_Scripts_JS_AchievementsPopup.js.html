1 "use strict";
  2 /**
  3  * @class AchPopup
  4  * @public
  5  */
  6 var AchPopup = AchPopup || {};
  7 /**
  8  * @memberOf AchPopup
  9  * @public
 10  */
 11 AchPopup.showAchievementWidget = function AchWidgetPopup(Achievementid)
 12 {
 13   // Let's populate the little widget
 14   var PopupUnlock = $('#PopupUnlock');
 15   PopupUnlock.find('img').prop('src', achManager.achievements[Achievementid].unlockedIconUrl);
 16   PopupUnlock.find('#achName').text(achManager.achievements[Achievementid].name);
 17   PopupUnlock.find('#achDescrip').text(achManager.achievements[Achievementid].description);
 18   PopupUnlock.css({'top': '300px', 'opacity': '1.0'});
 19   PopupUnlock.show();
 20   PopupUnlock.delay(1000).animate({top: 60, opacity: 0.2}, 500, function() {
 21   PopupUnlock.hide();
 22   });
 23 };
