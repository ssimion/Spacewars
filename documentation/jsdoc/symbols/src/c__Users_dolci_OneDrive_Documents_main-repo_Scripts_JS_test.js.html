 38 /** Test if two decks are different (They should be the same size)
 39  * @param array1 : first array
 40  * @param array2 : second array
 41  * @return {boolean}: false if array is shuffled
 42  */
 43 var IsShuffled = function (array1,array2) {
 44 	// @failsafe Return the shortest length to prevent overflow anyways.
 45 	var length = array1.length < array2.length ? array1.length : array2.length;  
 46 	for(var i = 0 ;i < length; ++i) {
 47 		if(array1[i] !== array2[i])
 48 			return false;
 49 	}
 50 	return true;
 51 }
