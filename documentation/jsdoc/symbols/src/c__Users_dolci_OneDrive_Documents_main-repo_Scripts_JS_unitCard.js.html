1 "use strict";
  2 
  3 /** Unit card class
  4  * @class UnitCard
  5  * @extend Card
  6  * @param {integer} health: the health of the card
  7  * @param {integer} attack: the attack of the card
  8  */
  9 var UnitCard = function(health,attack) {
 10 	this.nameOfCard = "Spaceship";
 11 	this.descriptionOfCard = "A little Spaceship";
 12 	this.uniqueID = Math.round(Math.random() * 1000000000000);
 13 	this.health = health || 0;
 14 	this.attack = attack || 1;
 15 };
 16 UnitCard.prototype = new Card();
 17 
 18 /** Returns the health of the unit card
 19  * @memberOf UnitCard
 20  * @public
 21  */
 22 UnitCard.prototype.GetHealth = function()
 23 {
 24 	return this.health;
 25 }
 26 /** Returns the attack of the unit card
 27  * @memberOf UnitCard
 28  * @public
 29  */
 30 UnitCard.prototype.GetAttack = function()
 31 {
 32 	return this.attack;
 33 }
 34 /** Override the to String Function
 35  * @memberOf UnitCard
 36  * @public
 37  */
 38 UnitCard.prototype.toString = function(){ 
 39 	return '[UnitCard "' + this.health + '"]';
 40 }
